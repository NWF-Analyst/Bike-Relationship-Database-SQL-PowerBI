// Before uploading it unto the SQLite database, I opted to visualize the Entity Relationship Diagram first in order to run it smoothly after uploading it

TABLE brands {
  Brand_ID integer [pk, increment] 
  Brand_Name text [not null]
}

TABLE categories {
  Category_ID integer [pk, increment]
  Category_Name text [not null]
}


TABLE customers {
  Customer_ID integer [pk, increment]
  First_Name text [not null]
  Last_Name text [not null]
  Phone_Number text
  Email text 
  Street text
  Zip_Code integer
}

TABLE order_items {
  Order_ID integer [pk, increment]
  Item_ID integer [unique]
  Product_ID integer [not null]
  Quantity text
  List_Price text 
  Discount text
}

TABLE orders { // This is the Fact Table
  Order_ID text [pk, increment]
  Customer_ID text [not null]
  Order_Status text [not null]
  Order_Date text
  Required_Date text 
  Shipped_Date date
  Store_ID integer
  Staff_ID integer
}

TABLE products {
  Product_ID text [pk, increment]
  Product_Name text [not null]
  Brand_ID text [not null]
  Category_ID text
  Model_Year text 
}

TABLE Staffs {
  Staff_ID text [pk, increment]
  First_Name text [not null]
  Last_Name text [not null]
  Email text
}

TABLE Stocks {
  Store_ID text [pk, increment]
  Product_ID text [not null]
  Quantity numeric  
}

TABLE Stores {
  Store_ID text [pk, increment]
  Store_Name text [not null]
  Phone text [not null]
  Email text
  Street text
  City text
  State Varchar(20)
  Zip_Code text
}
